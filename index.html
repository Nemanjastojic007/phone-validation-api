Integration Examples

            
Quick code snippets to get you started with our API.


            
Node.js (using axios)

            

                

                    const axios = require('axios');

async function validatePhone(phoneNumber) {
  try {
    const response = await axios.get(
      'https://your-api-domain.com/api/validate',
      {
        params: {
          phone: phoneNumber,
          country: 'US'
        },
        headers: {
          'x-api-key': 'your-api-key-here'
        }
      }
    );
    
    console.log(response.data);
    // {
    //   valid: true,
    //   number: "+12345678901",
    //   country: "US",
    //   type: "mobile"
    // }
  } catch (error) {
    console.error('Validation failed:', error.response?.data);
  }
}

validatePhone('+1234567890');

JavaScript (Fetch API)

            

                

                    async function validatePhone(phoneNumber) {
  try {
    const response = await fetch(
      `https://your-api-domain.com/api/validate?phone=${encodeURIComponent(phoneNumber)}&country=US`,
      {
        method: 'GET',
        headers: {
          'x-api-key': 'your-api-key-here',
          'Content-Type': 'application/json'
        }
      }
    );
    
    const data = await response.json();
    console.log(data);
  } catch (error) {
    console.error('Validation failed:', error);
  }
}

validatePhone('+1234567890');
